#SBATCH --nodes=1 #No. computers requested

#SBATCH --mem-per-cpu=2000mb #Per processor memory requested
#SBATCH --array=1-2%4
#SBATCH --time=00:01:00 #Walltime

module load intel/2017

for i in $(seq 1 16)
do
    MULTIPLIER="$((${SLURM_ARRAY_TASK_ID} - 1))"
    FIRST_COUNT="$((${SLURM_CPUS_ON_NODE} * ${MULTIPLIER}))"
    ID="$((${FIRST_COUNT} + ${i}))"
    $AMBERHOME/bin/sander -O -i ${ID}/nasqm_abs_${ID}.in -o ${ID}/nasqm_abs_${ID}.out -c ${ID}/ground_snap.${ID} -p m1.prmtop -r ${ID}/nasqm_abs_${ID}.rst -x ${ID}/nasqm_abs_${ID}.nc &
done
wait
